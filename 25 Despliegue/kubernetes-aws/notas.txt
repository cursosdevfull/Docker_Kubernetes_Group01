Instalar
- aws-cli
- eksctl
- helm

Para setear credenciales
aws configure

Para crear un cluster
eksctl create cluster --name test-cluster --without-nodegroup --region us-east-2 --zones us-east-2b,us-east-2a

Para crear nodos
eksctl create nodegroup \
--cluster test-cluster \
--version auto \
--name test-workers \
--node-type t3.medium \
--node-ami auto \
--nodes 1 \
--nodes-min 1 \
--nodes-max 3 \
--asg-access
